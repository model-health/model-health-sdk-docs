export const ZoomableImage = ({ src, alt, caption }) => {
  const [open, setOpen] = React.useState(false);
  return (
    <>
      <div style={{textAlign: 'center'}}>
        <img
          src={src} alt={alt}
          onClick={() => setOpen(true)}
          style={{width: '80%', cursor: 'zoom-in', borderRadius: '6px'}}
        />
        {caption && <p style={{fontSize: '0.8rem', color: 'var(--ifm-color-secondary-darkest)', marginTop: '0.4rem'}}>{caption}</p>}
      </div>
      {open && (
        <div onClick={() => setOpen(false)} style={{position: 'fixed', inset: 0, background: 'rgba(0,0,0,0.85)', display: 'flex', alignItems: 'center', justifyContent: 'center', zIndex: 9999, cursor: 'zoom-out', padding: '2rem'}}>
          <img src={src} alt={alt} style={{maxWidth: '90vw', maxHeight: '90vh', borderRadius: '8px', boxShadow: '0 8px 40px rgba(0,0,0,0.6)'}} />
        </div>
      )}
    </>
  );
};
