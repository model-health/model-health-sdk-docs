service.importSession(json, subject) { status ->
    when (status) {
        ImportStatus.CreatingSession           -> showCreating()
        is ImportStatus.CreatedSession         -> showCreated(status.sessionId)
        is ImportStatus.UploadingVideo         -> showUpload(status.trial, status.uploaded, status.total)
        ImportStatus.Processing                -> showProcessing()
    }
}
